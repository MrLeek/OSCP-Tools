# === PHASE 1: INSTANT WINS (run ALL before moving on) ===
# Who am I - SeImpersonate/SeAssignPrimaryToken = Potato family immediately
whoami /all
# Stored credentials - runas /savecred if found
#!cmd cmdkey /list
# OS info + patch level
systeminfo | findstr /B /C:"Host Name" /C:"OS Name" /C:"OS Version" /C:"System Type" /C:"Domain" /C:"Hotfix(s)"
# Users and admins
Get-LocalUser
Get-LocalGroupMember -Group "Administrators"
# Running processes + ports - cross-reference loopback listeners
Get-CimInstance Win32_Process | Select ProcessId,Name,CommandLine
Get-NetTCPConnection | Select LocalAddress,LocalPort,State,OwningProcess | Sort LocalPort
# Non-standard folders - recent timestamps = scheduled task activity
Get-ChildItem C:\ -Force
Get-ChildItem "C:\Program Files" -Force
Get-ChildItem "C:\Program Files (x86)" -Force
# Non-System32 services running as LocalSystem = high priority
Get-CimInstance Win32_Service | Select Name,StartName,State,PathName | Where-Object {$_.PathName -notmatch '\\System32\\'}
# AV check
wmic /node:localhost /namespace:\\root\securitycenter2 path antivirusproduct get displayname

# === PHASE 5: SCHEDULED TASKS ===
Get-ScheduledTask | where {$_.TaskPath -notlike "\Microsoft*"} | ft TaskName,TaskPath,State
# Watch filesystem for recently created/modified files
Get-ChildItem C:\ -Recurse -File -ErrorAction SilentlyContinue | Where-Object {$_.CreationTime -gt (Get-Date).AddMinutes(-10) -and $_.FullName -notlike "C:\Windows\*"} | Select FullName,CreationTime | Sort CreationTime -Descending

# === PHASE 6: CREDENTIAL HUNTING ===
# PowerShell history
Get-Content (Get-PSReadlineOption).HistorySavePath
# KeePass, keys, sensitive docs
Get-ChildItem -Path C:\Users -Include *.txt,*.pdf,*.xls,*.docx,*.log,*.kdbx,*.pem,*.ppk -File -Recurse -ErrorAction SilentlyContinue
# Backup/old files outside Windows dir
Get-ChildItem -Path C:\ -Include *.bak,*.old,*.sql,*.dump -File -Recurse -ErrorAction SilentlyContinue | Where-Object {$_.FullName -notmatch '\\Windows\\'}
# Quick secret search
gci C:\Users -r -file -ea 0 | ? {$_.Extension -match '\.(txt|ini|cfg|xml|ps1)$'} | sls 'password|passwd|pwd|token|secret' -CaseSensitive:$false -List
# Registry credential hunting
#!cmd reg query "HKLM\SOFTWARE\Microsoft\Windows NT\Currentversion\Winlogon"
#!cmd reg query "HKCU\Software\SimonTatham\PuTTY\Sessions"
#!cmd reg query "HKLM\SOFTWARE\Microsoft\Windows NT\CurrentVersion\Winlogon" /v DefaultPassword
# Unattend/sysprep files
Get-ChildItem -Recurse c:\windows -Include Unattend.xml,sysprep.xml -File -Force -ErrorAction SilentlyContinue
# IIS / web configs
Get-ChildItem -Path c:\inetpub -Include web.config,connectionstrings.config -File -Recurse -Force -ErrorAction SilentlyContinue
# XAMPP configs
Get-ChildItem -Recurse C:\xampp -Include *.txt,*.ini -File -ErrorAction SilentlyContinue

# === PHASE 7: ACL & PATH ===
# Check PATH for writable directories
$env:PATH.Split(';') | ForEach-Object { if (Test-Path $_) { (Get-Acl $_).Access | Where-Object { $_.IdentityReference -match $env:USERNAME -and $_.FileSystemRights -match 'Write' } | Select @{N='Path';E={$_}},FileSystemRights } }
