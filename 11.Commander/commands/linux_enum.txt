# === PHASE 1: INSTANT WINS (run ALL before moving on) ===
# Who am I - dangerous groups: disk, shadow, adm, docker, lxd → immediate privesc
id && whoami && hostname
# What can I run as root - NOPASSWD = game over
sudo -l
# What's running as root right now
ps fauxww | grep root
# Internal listeners - loopback services often exploitable
ss -tulnp
# SUID binaries - check each against GTFOBins
find / -perm -4000 -type f 2>/dev/null
# Capabilities - cap_setuid / cap_dac_override = privesc
getcap -r / 2>/dev/null
# Cron jobs - writable scripts or wildcard injection
cat /etc/crontab
ls -la /etc/cron.*
# Can I write to passwd/shadow - STOP and escalate if yes
ls -la /etc/passwd /etc/shadow
# Writable sudoers
ls -la /etc/sudoers /etc/sudoers.d/* 2>/dev/null
# Kernel version - feed into linux-exploit-suggester
uname -a && cat /etc/os-release
# NFS exports - rw,no_root_squash = STOP and escalate
showmount -e 2>/dev/null

# === PHASE 2: AUTOMATED TOOLS ===
# LinPEAS - run early, not as a last resort
./linpeas.sh | tee /tmp/linpeas_out.txt
# LSE - gradual readable output
./lse.sh -l 1
# Quick parse of linpeas output after it runs
grep -E "\[!\]|\[WARNING\]|SUDO|SUID|Capabilities|writable|CVE-" /tmp/linpeas_out.txt
grep -oE "CVE-[0-9]{4}-[0-9]+" /tmp/linpeas_out.txt | sort -u

# === PHASE 3: NETWORK & ENVIRONMENT ===
# Full network picture - look for dual-homed
ip a && ip r && arp -a
cat /etc/hosts
# PATH - writable dir early in PATH = hijack opportunity
echo $PATH | tr ':' '\n'
for p in $(echo $PATH | tr ':' ' '); do ls -ld $p 2>/dev/null; done
# Environment variables - may contain credentials or tokens
env

# === PHASE 4: USERS & CREDENTIALS ===
# All users with shells
cat /etc/passwd | grep -vE 'nologin|false'
# Group memberships - any non-standard groups
cat /etc/group | grep -v "^#"
# Shell histories - often contain leaked creds
cat ~/.bash_history 2>/dev/null
cat ~/.zsh_history 2>/dev/null
for u in $(cut -f1 -d: /etc/passwd); do echo "==$u=="; tail -20 /home/$u/.bash_history 2>/dev/null; done
# SSH keys
find /home /root /opt /var/www -type f \( -name "id_rsa*" -o -name "id_ed25519*" -o -name "*.pem" \) 2>/dev/null
# .env files and web configs
find / -name ".env" 2>/dev/null
find /var/www -name "config.php" -o -name "wp-config.php" 2>/dev/null

# === PHASE 5: INTERESTING FILES ===
# Recently modified files (potential cron targets)
find / -type f -mmin -10 -not -path "/proc/*" -not -path "/sys/*" 2>/dev/null
# World-writable files
find / -type f -writable -not -path "/proc/*" -not -path "/sys/*" 2>/dev/null | head -20
# Mail
ls /var/mail /var/spool/mail 2>/dev/null
# Non-standard SGID
find / -perm -2000 -type f 2>/dev/null

# === PHASE 6: PROCESS SNOOPING ===
# Transfer pspy to target first - catches hidden cron jobs, transient processes
# Run for 2-3 mins and look for: writable scripts, wildcards (*), relative paths, creds in args
./pspy64
# Manual check - root process environments (may contain tokens/credentials)
for p in $(pgrep -u root); do echo "PID $p"; tr '\0' '\n' < /proc/$p/environ 2>/dev/null | head -10; done
